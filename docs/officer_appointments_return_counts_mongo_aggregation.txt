db.getCollection("appointments").aggregate([
    {
        $match: { 'officer_id': 'OFFICER_ID' }
    },
    {
        $facet: {
            'inactive': [
                {
                    $match: {
                        $and: [
                            { 'data.resigned_on': { $exists: false } },
                            {
                                'company_status': {
                                    $in: [
                                        'dissolved',
                                        'removed',
                                        'converted-closed'
                                    ]
                                }
                            }
                        ]
                    }
                }
            ],
            'resigned': [
                {
                    $match: { 'data.resigned_on': { $exists: true } }
                }
            ]
        }
    },
    {
        $project: {
            'inactive_count': { $size: '$inactive' },
            'resigned_count': { $size: '$resigned' }
        }
    }
])
